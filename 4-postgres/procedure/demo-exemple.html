Exemple 1 : Procédure Simple pour Insérer un Enregistrement

Création d'une Table

CREATE TABLE employees (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100),
    salary NUMERIC
);

Procédure d'Insertion

CREATE OR REPLACE PROCEDURE insert_employee(emp_name VARCHAR, emp_salary NUMERIC)
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO employees (name, salary) VALUES (emp_name, emp_salary);
END;
$$;

Appel de la Procédure

CALL insert_employee('John Doe', 50000);

Exemple 2 : Procédure avec Gestion des Transactions

Procédure pour Transférer des Fonds Entre Comptes

Création des Tables
CREATE TABLE accounts (
    account_id SERIAL PRIMARY KEY,
    balance NUMERIC
);

INSERT INTO accounts (balance) VALUES (1000), (500);
Procédure de Transfert de Fonds
CREATE OR REPLACE PROCEDURE transfer_funds(sender INTEGER, receiver INTEGER, amount NUMERIC)
LANGUAGE plpgsql
AS $$
BEGIN
    BEGIN
        -- Déduire le montant du compte de l'expéditeur
        UPDATE accounts SET balance = balance - amount WHERE account_id = sender;

        -- Ajouter le montant au compte du destinataire
        UPDATE accounts SET balance = balance + amount WHERE account_id = receiver;

        -- Vérifier si les mises à jour sont réussies
        IF NOT FOUND THEN
            -- Si une mise à jour échoue, annuler la transaction
            ROLLBACK;
            RAISE EXCEPTION 'Transaction failed';
        END IF;

        -- Confirmer la transaction
        COMMIT;
    EXCEPTION WHEN OTHERS THEN
        -- En cas d'erreur, annuler la transaction
        ROLLBACK;
        RAISE;
    END;
END;
$$;
Appel de la Procédure
CALL transfer_funds(1, 2, 200);

Exemple 3 : Procédure avec Paramètres IN, OUT et INOUT

Procédure pour Calculer la Surface et le Périmètre d'un Rectangle

CREATE OR REPLACE PROCEDURE calculate_area_perimeter(
    length NUMERIC,
    width NUMERIC,
    OUT area NUMERIC,
    INOUT perimeter NUMERIC
)
LANGUAGE plpgsql
AS $$
BEGIN
    area := length * width;
    perimeter := 2 * (length + width);
END;
$$;

Appel de la Procédure

DO $$
DECLARE
    a NUMERIC;
    p NUMERIC := 0;
BEGIN
    CALL calculate_area_perimeter(5, 3, a, p);
    RAISE NOTICE 'Area: %, Perimeter: %', a, p;
END;
$$;

Exemple 4 : Procédure avec Cursor et Boucles

Création d'une Table des Ventes

CREATE TABLE sales (
    sale_id SERIAL PRIMARY KEY,
    product_name VARCHAR(100),
    sale_amount NUMERIC
);

INSERT INTO sales (product_name, sale_amount) VALUES
('Product A', 100),
('Product B', 150),
('Product C', 200);

Procédure pour Traiter les Ventes

CREATE OR REPLACE PROCEDURE process_sales()
LANGUAGE plpgsql
AS $$
DECLARE
    sale_record RECORD;
    cur_sales CURSOR FOR SELECT sale_id, product_name, sale_amount FROM sales;
BEGIN
    OPEN cur_sales;
    LOOP
        FETCH cur_sales INTO sale_record;
        EXIT WHEN NOT FOUND;
        RAISE NOTICE 'Processing Sale ID: %, Product: %, Amount: %', sale_record.sale_id, sale_record.product_name, sale_record.sale_amount;
    END LOOP;
    CLOSE cur_sales;
END;
$$;

Appel de la Procédure

CALL process_sales();

Exemple 5 : Procédure avec Gestion des Erreurs Complexe

Procédure pour Ajouter un Employé et Gérer les Erreurs

CREATE OR REPLACE PROCEDURE add_employee(name VARCHAR, salary NUMERIC)
LANGUAGE plpgsql
AS $$
BEGIN
    BEGIN
        INSERT INTO employees (name, salary) VALUES (name, salary);
        COMMIT;
    EXCEPTION WHEN unique_violation THEN
        ROLLBACK;
        RAISE NOTICE 'Employee with name % already exists', name;
    EXCEPTION WHEN others THEN
        ROLLBACK;
        RAISE NOTICE 'An unexpected error occurred';
    END;
END;
$$;

Appel de la Procédure

CALL add_employee('Jane Doe', 60000);
